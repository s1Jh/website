color_t sample_surface(SDL_Surface* s, int x, int y)
{
    auto get_component = [](
        uint32_t sample,
        uint32_t mask,
        uint8_t shift,
        uint8_t loss
    ) -> uint8_t
    {
        sample &= mask;
        sample >>= shift;
        sample <<= loss;

        return (uint8_t)sample;
    };

    uint32_t raw_sample;
    color_t c;

    SDL_LockSurface(s);

    uint32_t* data = (uint32_t*) s->pixels;

    x = std::abs(x) % s->w;
    y = std::abs(y) % s->h;

    raw_sample = data[x + y * s->w];

    c.r = get_component(
        raw_sample,
        s->format->Rmask,
        s->format->Rshift,
        s->format->Rloss
    );
    c.g = get_component(
        raw_sample,
        s->format->Gmask,
        s->format->Gshift,
        s->format->Gloss
    );
    c.b = get_component(
        raw_sample,
        s->format->Bmask,
        s->format->Bshift,
        s->format->Bloss
    );
    if (s->format->Amask == 0)
        c.a = 0xff;
    else
        c.a = get_component(
            raw_sample,
            s->format->Amask,
            s->format->Ashift,
            s->format->Aloss
        );

    SDL_GetRGBA(raw_sample, s->format, &c.r, &c.g, &c.b, &c.a);
    SDL_UnlockSurface(s);

    return c;
}
